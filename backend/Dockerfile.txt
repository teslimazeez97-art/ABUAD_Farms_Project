# backend/Dockerfile
FROM node:18-slim

# Install netcat for waiting on DB in entrypoint
RUN apt-get update && apt-get install -y netcat && rm -rf /var/lib/apt/lists/*

WORKDIR /usr/src/app

# Copy package metadata first for caching
COPY package*.json ./

# Install dependencies
RUN npm ci --only=production --no-audit --no-fund || npm install

# Copy app source
COPY . .

# Add entrypoint and make executable (chmod here ensures Windows-created file is executable in container)
COPY docker-entrypoint.sh /usr/src/app/docker-entrypoint.sh
RUN chmod +x /usr/src/app/docker-entrypoint.sh

EXPOSE 5000

ENTRYPOINT ["./docker-entrypoint.sh"]
